import java.util.*;
public class Main {
    public static void main(String args[]) {
        // Your Code Here
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int[] arr = new int[n];
		for(int i = 0 ; i < n ; i++){
			arr[i] = sc.nextInt();
		}
		reverseAnArray(arr, n - 1, 0);
    }
	private static void reverseAnArray(int[] arr, int n, int curr){
		if (curr > n){
			return;
		}
		reverseAnArray(arr,  n, curr+1);
		System.out.print(arr[curr] + " ");

	}
}
